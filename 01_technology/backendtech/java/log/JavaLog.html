# Java日志

# 1. log4j编程式实现

import org.apache.log4j.*;

/**
 * @Title: LogProcessor
 * @Copyright (C) 2022
 * @Description: 初始化日志体系
 * @Author: jofen
 * @Revision History:
 * @Revision 1.0 
 */
public class LogProcessor {
    /**
     * 初始化日志体系
     * @param log
     */
    public void process(LoggerConfig log) {
        PatternLayout layout = new PatternLayout();
        String conversionPattern = log.getConversionPattern();
        layout.setConversionPattern(conversionPattern);

        ConsoleAppender consoleAppender = new ConsoleAppender();
        consoleAppender.setLayout(layout);
        // 默认DEBUG级别
        consoleAppender.setThreshold(Level.toLevel(log.getLevelConsole(), Level.DEBUG));
        consoleAppender.activateOptions();

        FileAppender fileAppender = new FileAppender();
        fileAppender.setFile(log.getFile());
        fileAppender.setLayout(layout);
        // 默认INFO级别
        fileAppender.setThreshold(Level.toLevel(log.getLeveFile(), Level.INFO));
        fileAppender.activateOptions();

        Logger rootLogger = Logger.getRootLogger();
        //rootLogger.setLevel();
        rootLogger.addAppender(consoleAppender);
        rootLogger.addAppender(fileAppender);
    }

    public static void main(String[] args) {
        CommConfig config = new YamlProcessor().process(CommConfig.class);
        System.out.println(config);

        new LogProcessor().process(config.getLog());

        Logger log = Logger.getLogger(LogProcessor.class);
        log.debug("debug level's log");
        log.info("info level's log");
    }
}
