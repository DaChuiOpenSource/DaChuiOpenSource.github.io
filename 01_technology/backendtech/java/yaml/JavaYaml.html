# 2.1. Java-SnakeYAML

package org.lyqc.splitxls.core;

import org.apache.log4j.Logger;
import org.yaml.snakeyaml.Yaml;

import java.io.InputStream;

/**
 * @Title: YamlProcessor
 * @Copyright (C) 2022
 * @Description:
 * @Author: 04012
 * @Revision History:
 * @Revision 1.0 2022-3-19
 */
public class YamlProcessor {
    private static final Logger log = Logger.getLogger(YamlProcessor.class);
    /**
     * 默认yaml文件进行加载
     * @param type
     * @param <T>
     * @return
     */
    public <T> T process(Class<T> type) {
        return this.process("split.yaml", type);
    }

    /**
     * 指定yaml文件进行加载
     * @param yamlFile
     * @param type
     * @param <T>
     * @return
     */
    public <T> T process(String yamlFile, Class<T> type) {
        Yaml yaml = new Yaml();
        InputStream inputStream = this.getClass().getClassLoader()
                .getResourceAsStream(yamlFile);

        T t = (T)yaml.loadAs(inputStream, type);
        return t;
    }

    public static void main(String[] args) {

        //CommConfig config = new YamlProcessor().process(CommConfig.class);
        //ShareConfig config = new YamlProcessor().process("share.yaml", ShareConfig.class);
        PrecheckConfig config = new YamlProcessor().process("precheck.yaml", PrecheckConfig.class);
        System.out.println(config);
        log.info(config);
    }
}
