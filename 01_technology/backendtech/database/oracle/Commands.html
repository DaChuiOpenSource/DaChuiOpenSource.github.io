--change password

# 8. 检查是否有连接

select count(1) from gv$session where username = 'USERNAME';

# 9. 查看用户下数据大小

select sum(BYTES/1024/1024/1024) from dba_segments where owner='USERNAME';

# 10. 检查、创建、授权目录

select directory_name,DIRECTORY_PATH from all_directories where directory_name='DPDIR';
create directory dpdir as '/home/oracle/dpdir';
grant read,write on directory dpdir to username;
