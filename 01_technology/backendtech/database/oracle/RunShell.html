# Oracle关于Shell执行语句

# 1. 执行方式

sqlplus -s / as sysdba <<EOF | tee run.log
set lines 300
prompt 检查待用的数据库表空间TBS是否存在:
SELECT 1 FROM dba_tablespaces t where t.TABLESPACE_NAME = 'TBS';
prompt
prompt 检查数据库表空间大小及使用情况:
column TABLESPACE_NAME format A25;
column TS_SIZE format 999999999;
SELECT t.tablespace_name, round(SUM(bytes / (1024 * 1024)), 0) ts_size FROM dba_tablespaces t, dba_data_files d
WHERE t.tablespace_name = d.tablespace_name GROUP BY t.tablespace_name;
prompt
select dbf.tablespace_name,dbf.totalspace as "totalspace(M)",dbf.totalblocks,dfs.freespace,dfs.freeblocks,(1-(dfs.freespace/dbf.totalspace))*100 as usedPercent,
       dbf.maxbytes as "maxspace(M)"
from (select t.tablespace_name,sum(t.bytes)/1024/1024 totalspace,sum(t.blocks) totalblocks,sum(t.maxbytes)/1024/1024 as maxbytes from dba_data_files t group by t.tablespace_name) dbf,
     (select tt.tablespace_name,sum(tt.bytes)/1024/1024 freespace,sum(tt.blocks) freeblocks from dba_free_space tt group by tt.tablespace_name) dfs
where trim(dbf.tablespace_name)=trim(dfs.tablespace_name);
prompt
prompt 检查数据库字符集:
column NLS_CHARACTERSET format A20;
select a.VALUE AS NLS_CHARACTERSET from nls_database_parameters a where a.PARAMETER='NLS_CHARACTERSET';
prompt 检查字符计算单位:
column DISPLAY_VALUE format A20;
select a.DISPLAY_VALUE from v\$parameter a where a.NAME = 'nls_length_semantics';
prompt 检查中间库字符集:
column PARAMETER format A50;
column VALUE format A50;
SELECT * FROM NLS_DATABASE_PARAMETERS ORDER BY PARAMETER;
EOF

# 2. 获取返回值

connTestResult='sqlplus -S $username/$(password)@$ipstr:$portNumber/$serviceName<<EOF
    set heading off;
    set pagesize 0;
    set feedback off;
    set echo off;
    select 1 from dual;
    exit;
EOF
if [[ $connTestResult -eq 1 ]]; then
    echo "connect ok"
else 
    echo "test db connect error"
    exit 1;
fi
