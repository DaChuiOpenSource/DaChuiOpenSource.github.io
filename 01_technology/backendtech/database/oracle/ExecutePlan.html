# Oracle语句优化之执行计划

# 1. 查看执行计划

以下操作尽量在命令窗口执行,plsql执行的话可能没有预期的字段数据

# 1.1. dbms_xplan.display_cursor

--设置环境及变量
show parameter statistics_le;
alter session set statistics_level=all;
set pagesize 150;
set linesize 250;
set serveroutput off;

--执行sql
select /*+ jx1 */ from bus_prd_info;

--获取sql_id
select sql_id,child_number,sql_text from v$sql where upper(sql_text) like upper('%jx1%') and upper(sql_text) not like '%FROM V$SQL WHERE UPPER(SQL_TEXT) LIKE %';

--查看实际的执行计划
select * from table(dbms_xplan.display_cursor('4rya72pd7ghc1','0','ALLSTATS LAST'));
