# Oracle数据库备份与恢复

# 1. 备份

--数据泵
expdp "'/ as sysdba'" directory=dpdir dumpfile=schema_qiyitest.dmp logfile=exp_-schema_qiyitest.log compression=all schemas=schemaname
select username from all_users where common='NO';

--linux exp、imp
exp user/passwd file=/home/oracle/dpdir/20180921_user.dmp owner=user log=/home/oracle/dpdir/exp_20180921_user.log

--windows exp、imp
exp user/passwd@ipaddress:1521/sid file=%cd%\20181008_user.dmp owner=user log=%cd%\exp_20181008_user.log

# 2. 恢复

--数据泵
impdp "'/ as sysdba'" directory=dpdir dumpfile=lczg_201910310928.dmp remap_schema=schemaname:schemaothername logfile=impdp_`date +%Y%m%d%H%M%S`.log
impdp "'/ as sysdba'" directory=dpdir cluster=no remap_schema=schemaname:schemaothername dumpfile=schema_qiyitest.dmp logfile=schema_qiyitest-imp.log
select username from all_users where common='NO';

--linux exp、imp
imp user/passwd file=/home/oracle/dpdir/20181008_user.dmp full=y ignore=y log=/home/oracle/dpdir/imp_20181008_userlog

--windows exp、imp
imp user/passwd@ipaddress:1521/sid file=%cd%\20181008_user.dmp full=y ignore=y
