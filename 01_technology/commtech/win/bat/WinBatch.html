# Window批处理

# 1. 循环接收输入

@echo off

:again
echo.
echo.*************************************************
echo.请选择操作：
echo.1  - 操作1
echo.2  - 操作2
echo.3  - 操作3
echo.4  - 操作4
echo.5  - 操作5
echo.6  - 操作6
echo.0  - 操作1
echo.*************************************************

set /p InputNum=
set flag=false
if %InputNum%==1 set flag=true
if %InputNum%==2 set flag=true
if %InputNum%==3 set flag=true
if %InputNum%==4 set flag=true
if %InputNum%==5 set flag=true
if %InputNum%==6 set flag=true
rem if %InputNum%==0 set flag=false

rem echo %InputNum% %flag%
if %flag% == true (
    goto doaction
) else (
    goto dono
)

:doaction
rem jdk1.8.0_172\bin\java.exe -Dfile.encoding=UTF-8 -classpath .;.\lib\* -XX:PermSize=256m -XX:MaxPermSize=512m -Xms1024m -Xmx1024m com.combat.core.Engine
jdk1.8.0_172\bin\java.exe -Dfile.encoding=GBK -classpath .;.\conf;.\lib\* -XX:PermSize=256m -XX:MaxPermSize=512m -Xms1024m -Xmx1024m org.Process %InputNum%
echo.
set InputNum=0
goto again

:dono
pause
