# 2. 目录下文件数

统计当前目录下文件数（包含子目录）及首层子目录下文件数（包含子目录）

#!/bin/bash

echo `pwd` `ls -lR | grep '^-' | wc -l`
for perfile in `ls`
do
    if [ -d $perfile ];
    then
        echo `pwd`/$perfile `ls -lR $perfile | grep '^-' | wc -l`
    fi
done
